Communicating with the Messaging Infrastructure Using HTTP

This document outlines the process of communicating with the messaging infrastructure using HTTP to send messages synchronously and asynchronously.

Messaging Infrastructure Web Application

In order for a client to communicate with a queuing system such as ActiveMQ, the messaging infrastructure has to be deployed to a servlet container such as Tomcat.

The web application has a SenderServlet class mapped to /msginf/sender which is used to communicate with the messaging infrastructure. Clients send requests to this servlet and parse the response body.

The HTTP request parameters are:

HTTP Request Parameter Description
messagingSystem The name of the messaging system to use as defined in the msginf.xml file, e.g. activemq.
connector The name of the connector defined in the msginf.xml file the message is being sent to.
data The XML data.

The response body is sent back as text/plain and contains the following for each connector type (submit or request/reply):

Scenario Response
Asynchronous (submit) "Message sent successfully."
Synchronous (request/reply) The XML reply from the application the message was sent to.

Follow the instructions here to build the messaging infrastructure war file msginf.war.

Deploy the msginf.war file to the servlet container.

Test Code

The following is a simple Java class which tests the messaging infrastructure:

package nz.co.pukekocorp.msginf.client.adapter.spring;

import org.apache.commons.httpclient.HttpClient;
import org.apache.commons.httpclient.HttpStatus;
import org.apache.commons.httpclient.methods.PostMethod;

public class TestSenderServlet {
    private HttpClient httpClient;
    private String url = "http://localhost:8080/msginf/sender";
    
    public TestSenderServlet() {
        httpClient = new HttpClient();
    }

    private String createMessage() {
        String message = "<?xml version=\"1.0\"?>" +
                         "<Data xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xmlns=\"http://localhost/test\">" +
                         "  <MessageID>12345</MessageID>" +
                         "  <Claim>" +
                         "    <Name>Fred Dagg</Name>" +
                         "    <Name>Bart Simpson</Name>" +
                         "  </Claim>" +
                         "</Data>";
        return message;
    }

    public String postSubmitMessage() {
        String reply = "";
        PostMethod post = null;
        String message = createMessage();
        try {
            post = new PostMethod(url);
            post.setParameter("messagingSystem", "activemq");
            post.setParameter("connector", "SubmitConnector");
            post.setParameter("data", message);
            int res = httpClient.executeMethod(post);
            if (res != HttpStatus.SC_OK) {
                System.err.println("Method failed: " + post.getStatusLine());
            }
            reply = post.getResponseBodyAsString();        
        } catch (Exception e) {
            e.printStackTrace();
         } finally {
          // Release the connection.
          post.releaseConnection();
        }
         return reply;
    }

    public String postRequestReplyMessage() {
        String reply = "";
        PostMethod post = null;
        String message = createMessage();
        try {
            post = new PostMethod(url);
            post.setParameter("messagingSystem", "activemq");
            post.setParameter("connector", "RequestReplyConnector");
            post.setParameter("data", message);
            int res = httpClient.executeMethod(post);
            if (res != HttpStatus.SC_OK) {
                System.err.println("Method failed: " + post.getStatusLine());
            }
            reply = post.getResponseBodyAsString();        
        } catch (Exception e) {
            e.printStackTrace();
         } finally {
          // Release the connection.
          post.releaseConnection();
        }
         return reply;
    }

    public void postResetCountMessage() {
        PostMethod post = null;
        String message = createMessage();
        try {
            post = new PostMethod(url);
            post.setParameter("messagingSystem", "activemq");
            post.setParameter("connector", "RequestReplyConnector");
            post.setParameter("data", "XXXXXXXXXXX");
            post.setParameter("resetCount", "true");
            int res = httpClient.executeMethod(post);
            if (res != HttpStatus.SC_OK) {
                System.err.println("Method failed: " + post.getStatusLine());
            }
        } catch (Exception e) {
            e.printStackTrace();
         } finally {
          // Release the connection.
          post.releaseConnection();
       }
    }

    public static void main(String[] args) {
        TestSenderServlet test = new TestSenderServlet();
        for (int i = 0; i < 10; i++) {
            System.out.println(test.postSubmitMessage());
        }
        for (int i = 0; i < 10; i++) {
            System.out.println(test.postRequestReplyMessage());
        }
        test.postResetCountMessage();
    }
}
